### 1.5 测试原则

| 原则 | 说明 | 实践方法 |
|------|------|---------|
| **分层测试** | Python负责采集写入测试，Java负责查询测试 | 明确职责分离 |
| **真实性** | 使用真实数据库，验证数据真实写入 | 直连数据库实例 |
| **隔离性** | 测试数据独立，不污染生产环境 | 事务回滚/测试数据库 |
| **全链路** | 从AKShare到数据库完整穿透 | Python采集→数据库→Java查询 |
| **Mock外部** | AKShare API调用使用Mock | 避免依赖外部服务 |
| **并行测试** | Python和Java测试可独立执行 | 无相互依赖 |

#### 4.1.2 测试代码示例

```java
/**
 * 数据采集模块 - 完整数据流集成测试
 * 
 * 测试特性:
 * 1. 通过 application-test.yaml 直连远程 Docker 数据库实例
 * 2. @Transactional 自动回滚，保证测试隔离
 * 3. 完整数据流验证: AKTools API → Service → Mapper → Database
 * 4. 数据库直接查询验证，非Mock
 */
@Slf4j
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@Transactional
@ActiveProfiles("test")
@TestMethodOrder(MethodOrderer.OrderAnnotation.class)
class DataCollectionIntegrationTest extends BaseIntegrationTest {

    @Autowired
    private StockDataService stockDataService;

    @Autowired
    private StockInfoMapper stockInfoMapper;

    @Autowired
    private StockPricesRepository stockPricesRepository;

    @Autowired
    private JdbcTemplate jdbcTemplate;

    @Autowired
    private MongoTemplate mongoTemplate;

    // ==================== V1: 数据源验证 ====================

    @Test
    @Order(1)
    void testV1_AKToolsAPIResponse() {
        log.info("=== V1: AKTools API响应验证 ===");
        
        // 调用Service获取股票列表
        List<StockInfo> stockList = stockDataService.fetchStockList();
        
        // 验证API响应正确
        assertThat(stockList).isNotNull();
        assertThat(stockList.size()).isGreaterThan(4000);
        
        // 验证字段完整性
        StockInfo firstStock = stockList.get(0);
        assertThat(firstStock.getCode()).isNotNull();
        assertThat(firstStock.getName()).isNotNull();
        assertThat(firstStock.getMarket()).isNotNull();
        
        log.info("V1 通过: AKTools API响应正确, 股票数量: {}", stockList.size());
    }

    // ==================== V2: Service处理验证 ====================

    @Test
```

```python
from sqlalchemy import text

class TestMySQLValidation:
    """MySQL数据验证"""
    
    def assert_data_exists(self, db_session, table, condition):
        """验证数据存在"""
        result = db_session.execute(
            text(f"SELECT COUNT(*) FROM {table} WHERE {condition}")
        ).scalar()
        assert result > 0, f"数据应存在于{table}表"
    
    def assert_row_count(self, db_session, table, expected_count):
        """验证记录数"""
        count = db_session.execute(
            text(f"SELECT COUNT(*) FROM {table}")
        ).scalar()
        assert count == expected_count, f"记录数应为{expected_count}，实际{count}"
    
    def query_value(self, db_session, sql, params=None):
        """查询单个值"""
        result = db_session.execute(text(sql), params or {})
        return result.scalar()
    
    # 使用示例
    def test_validate_mysql_data(self, db_session):
        # 验证数据存在
        self.assert_data_exists(db_session, "stock_info", "code = '000001'")
        
        # 验证记录数
        self.assert_row_count(db_session, "stock_info", 5000)
        
        # 验证字段值
        name = self.query_value(
            db_session,
```
