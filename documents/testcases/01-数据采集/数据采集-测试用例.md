/**
 * 数据采集E2E测试 - 完整业务流程穿透测试
 */
@Slf4j
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@ActiveProfiles("test")
class DataCollectionE2ETest {

    @Autowired
    private StockDataService stockDataService;

    @Autowired
    private StockInfoMapper stockInfoMapper;

    @Autowired
    private StockPricesRepository stockPricesRepository;

    @Autowired
    private JdbcTemplate jdbcTemplate;

    @LocalServerPort
    private int port;

    @Test
    void testFullFlow_DataCollectionToStorage() {
        log.info("=== E2E测试: 数据采集到存储完整流程 ===");
        
        // ========== 阶段1: 同步股票列表 ==========
        log.info("阶段1: 同步股票列表到MySQL");
        List<StockInfo> stockList = stockDataService.fetchStockList();
        assertThat(stockList).isNotEmpty();
        
        // 保存到MySQL
        for (StockInfo stock : stockList) {
            stock.setCreateTime(new Date());
            stock.setDeleted("0");
            stockInfoMapper.insert(stock);
        }
        
        // 验证MySQL
        Integer stockCount = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM stock_info", 
            Integer.class
        );
        assertThat(stockCount).isGreaterThan(0);
        log.info("MySQL股票数据: {} 条", stockCount);
        
        // ========== 阶段2: 同步历史价格 ==========

        // 获取一只股票同步历史数据

        stockDataService.syncStockHistory(stockCode, 60);

        assertThat(prices).isNotEmpty();

        // ========== 阶段3: 验证数据完整性 ==========
        log.info("阶段3: 验证数据完整性");
        
        // 验证股票数据
        StockInfo savedStock = stockInfoMapper.getByCode(stockCode);
        assertThat(savedStock).isNotNull();
        assertThat(savedStock.getCode()).isEqualTo(stockCode);
        
        // 验证价格数据
        assertThat(prices.get(0).getCode()).isEqualTo(stockCode);
        assertThat(prices.get(0).getPrice1()).isNotNull();
        
        log.info("✅ E2E测试通过: 数据采集 → MySQL/MongoDB 完整流程验证");
    }

/**
 * 数据采集性能测试
 */
@Test
void testPerformance_BatchStockSync() {
    // 准备批量数据
    List<StockInfo> batchData = new ArrayList<>();
    for (int i = 0; i < 100; i++) {
        batchData.add(StockInfo.builder()
                .code("PERF" + String.format("%04d", i))
                .name("性能测试股票" + i)
                .market("SZ")
                .isTradable(1)
                .createTime(new Date())
                .deleted("0")
                .build());
    }
    
    // 测量执行时间
    long startTime = System.currentTimeMillis();
    
    // 执行批量写入
    for (StockInfo stock : batchData) {
        stockInfoMapper.insert(stock);
    }
    
    long endTime = System.currentTimeMillis();
    long duration = endTime - startTime;
    
    // 验证性能
    assertThat(duration).as("批量写入应在1000ms内完成").isLessThan(1000);
    
    // 验证数据
    Integer count = jdbcTemplate.queryForObject(
        "SELECT COUNT(*) FROM stock_info WHERE code LIKE 'PERF%'", 
        Integer.class
    );
    assertThat(count).isEqualTo(100);
    
    log.info("批量写入100条数据耗时: {}ms", duration);
}
