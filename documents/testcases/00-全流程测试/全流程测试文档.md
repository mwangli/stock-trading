## 2. 设计原则

### 2.1 核心理念

**真实验证，拒绝 Mock** - 所有测试必须验证数据真实写入数据库，确保从数据源到数据终点的完整链路。

### 2.2 测试原则

| 原则 | 说明 |
|------|------|
| **真实性** | 使用真实数据库，验证数据真实写入 |
| **隔离性** | 测试数据自动回滚，不污染生产环境 |
| **可追溯** | 每个测试步骤都可验证数据状态 |
| **全链路** | 从 API 到数据库完整穿透 |
| **可重复** | 测试可多次执行，结果一致 |

## 5. 测试基类设计

### 5.1 集成测试基类

```java
/**
 * 集成测试基类 - 真实数据库测试
 * 
 * 特性:
 * 1. 通过 application-test.yaml 直连远程 Docker 数据库实例
 * 2. @Transactional 自动回滚，不污染数据库
 * 3. 提供数据库验证工具方法
 */
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@Transactional
@ActiveProfiles("test")
@TestMethodOrder(MethodOrderer.OrderAnnotation.class)
public abstract class BaseIntegrationTest {

    @Autowired
    protected JdbcTemplate jdbcTemplate;

    @Autowired
    protected TestRestTemplate restTemplate;

    @Autowired
    protected MongoTemplate mongoTemplate;

    // ==================== 数据库验证工具方法 ====================

    /**
     * 验证 MySQL 表中数据存在
     */
    protected void assertDataExists(String table, String whereClause) {
        Integer count = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM " + table + " WHERE " + whereClause,
            Integer.class
        );
        assertThat(count).as("数据应该存在: " + whereClause).isGreaterThan(0);
    }

    /**
     * 验证 MySQL 表中数据不存在
     */
    protected void assertDataNotExists(String table, String whereClause) {
        Integer count = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM " + table + " WHERE " + whereClause,
            Integer.class
        );
        assertThat(count).as("数据不应该存在: " + whereClause).isEqualTo(0);
    }

    /**
     * 验证 MySQL 表记录数
     */
    protected void assertRowCount(String table, int expectedCount) {
        Integer count = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM " + table,
            Integer.class
        );
        assertThat(count).isEqualTo(expectedCount);
    }

    /**
     * 查询单个字段值
     */
    protected <T> T queryValue(String sql, Class<T> type, Object... args) {
        return jdbcTemplate.queryForObject(sql, type, args);
    }

    /**
     * 验证 MongoDB 集合中数据存在
     */
    protected void assertMongoDataExists(String collection, String field, Object value) {
        Query query = Query.query(Criteria.where(field).is(value));
        long count = mongoTemplate.count(query, collection);
        assertThat(count).as("MongoDB数据应该存在").isGreaterThan(0);
    }

    /**
     * 验证 MongoDB 集合记录数
     */
    protected void assertMongoRowCount(String collection, long expectedCount) {
        long count = mongoTemplate.getCollection(collection).countDocuments();
        assertThat(count).isEqualTo(expectedCount);
    }

    /**
     * 插入测试数据到 MySQL
     */
    protected void insertTestData(String table, Map<String, Object> data) {
        String columns = String.join(", ", data.keySet());
        String placeholders = data.keySet().stream().map(k -> "?").collect(Collectors.joining(", "));
        String sql = "INSERT INTO " + table + " (" + columns + ") VALUES (" + placeholders + ")";
        jdbcTemplate.update(sql, data.values().toArray());
    }

    /**
     * 等待异步操作完成
     */
    protected void awaitFor(int milliseconds) {
        try {
            Thread.sleep(milliseconds);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
}
```

### 5.2 端到端测试基类

```java
/**
 * 端到端测试基类 - 全链路穿透测试
 */
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@ActiveProfiles("test")
public abstract class BaseE2ETest {

    @LocalServerPort
    protected int port;

    protected RequestSpecification requestSpec;

    @BeforeEach
    void setUp() {
        RestAssured.port = port;
        requestSpec = new RequestSpecBuilder()
                .setContentType(ContentType.JSON)
                .setAccept(ContentType.JSON)
                .build();
    }

    /**
     * 发送 GET 请求并验证响应
     */
    protected Response getAndVerify(String path, int expectedStatus) {
        return given()
                .spec(requestSpec)
                .when()
                .get(path)
                .then()
                .statusCode(expectedStatus)
                .extract()
                .response();
    }

    /**
     * 发送 POST 请求并验证响应
     */
    protected Response postAndVerify(String path, Object body, int expectedStatus) {
        return given()
                .spec(requestSpec)
                .body(body)
                .when()
                .post(path)
                .then()
                .statusCode(expectedStatus)
                .extract()
                .response();
    }

    /**
     * 验证数据库最终一致性
     */
    protected void assertEventually(Runnable assertion, Duration timeout) {
        Awaitility.await()
                .atMost(timeout)
                .untilAsserted(() -> assertion.run());
    }
}
```

```java
@Transactional

class FullTradingFlowE2ETest extends BaseIntegrationTest {

    @Autowired

    private RiskControlService riskControlService;

    @Autowired
    private TradeExecutionService tradeService;

    @Autowired
    private JdbcTemplate jdbcTemplate;

    @Autowired
    private MongoTemplate mongoTemplate;

    // ==================== 阶段1: 数据准备 ====================

    @Test
    @Order(1)
    void stage1_prepareTestData() {
        log.info("=== 阶段1: 准备测试数据 ===");

        // 1.1 插入股票信息到 MySQL
        StockInfo stock = new StockInfo();
        stock.setCode("000001");
        stock.setName("平安银行");
        stock.setMarket("SZ");
        stock.setIsTradable(1);
        stock.setCreateTime(new Date());
        stock.setDeleted("0");
        stockInfoMapper.insert(stock);

        // 验证写入
        assertDataExists("stock_info", "code = '000001'");
        log.info("V1.1 通过: 股票信息写入MySQL");

        // 1.2 插入历史价格到 MongoDB
        for (int i = 0; i < 60; i++) {
            StockPrices price = new StockPrices();
            price.setCode("000001");
            price.setDate(LocalDate.now().minusDays(60 - i));
            price.setPrice1(10.0 + Math.random());
            price.setPrice2(10.5 + Math.random());
            price.setPrice3(9.8 + Math.random());
            price.setPrice4(10.2 + Math.random());
            price.setVolume(1000000L);
            stockPricesRepository.save(price);
        }

        // 验证写入
        assertMongoRowCount("stockPrices", 60);
        log.info("V1.2 通过: 历史价格写入MongoDB");
    }

    // ==================== 阶段2: 选股流程 ====================

    @Test
    @Order(2)
    void stage2_stockSelection() {
        log.info("=== 阶段2: 选股流程 ===");

    }

    @Test
    @Order(3)
    void stage3_decisionMaking() {
        log.info("=== 阶段3: 决策流程 ===");

        // 准备选股结果
        SelectionResult selection = new SelectionResult();
        selection.setSelectedStock("000001");
        selection.setOverallScore(75);

        // 3.1 生成决策
```

```java
/**
 * 数据流验证工具
 */
public class DataFlowVerifier {

    private final JdbcTemplate jdbcTemplate;
    private final MongoTemplate mongoTemplate;
    private final RedisTemplate<String, String> redisTemplate;

    /**
     * 验证 MySQL 数据流
     */
    public void verifyMySQLFlow(String table, String code, Map<String, Object> expectedValues) {
        Integer count = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM " + table + " WHERE code = ?",
            Integer.class, code
        );
        assertThat(count).as("数据应该存在").isGreaterThan(0);

        for (Map.Entry<String, Object> entry : expectedValues.entrySet()) {
            Object actual = jdbcTemplate.queryForObject(
                "SELECT " + entry.getKey() + " FROM " + table + " WHERE code = ?",
                Object.class, code
            );
            assertThat(actual).as("字段 " + entry.getKey()).isEqualTo(entry.getValue());
        }
    }

    /**
     * 验证 MongoDB 数据流
     */
    public void verifyMongoDBFlow(String collection, String code, int expectedCount) {
        Query query = Query.query(Criteria.where("code").is(code));
        long count = mongoTemplate.count(query, collection);
        assertThat(count).as("MongoDB数据数量").isEqualTo(expectedCount);
    }

    /**
     * 验证 Redis 数据流
     */
    public void verifyRedisFlow(String keyPattern, String expectedValue) {
        Set<String> keys = redisTemplate.keys(keyPattern);
        assertThat(keys).as("Redis键应该存在").isNotEmpty();
        
        for (String key : keys) {
            String value = redisTemplate.opsForValue().get(key);
            assertThat(value).as("Redis值").isEqualTo(expectedValue);
        }
    }

    /**
     * 打印数据流验证报告
     */
    public void printVerificationReport(String flowName, List<VerificationPoint> points) {
        log.info("========== {} 数据流验证报告 ==========", flowName);
        for (VerificationPoint point : points) {
            String status = point.isPassed() ? "通过" : "失败";
            log.info("  {} - {}: {}", point.getId(), point.getName(), status);
        }
        log.info("=========================================");
    }
}
```

### 9.2 并发测试

```java
@Test
void testConcurrentTradingFlow() throws InterruptedException {
    int threadCount = 10;
    ExecutorService executor = Executors.newFixedThreadPool(threadCount);
    CountDownLatch latch = new CountDownLatch(threadCount);
    AtomicInteger successCount = new AtomicInteger(0);

    for (int i = 0; i < threadCount; i++) {
        final int index = i;
        executor.submit(() -> {
            try {
                String stockCode = "00000" + (index + 1);
                executeFullTradingFlow(stockCode);
                successCount.incrementAndGet();
            } finally {
                latch.countDown();
            }
        });
    }

    latch.await(60, TimeUnit.SECONDS);
    
    assertThat(successCount.get()).isGreaterThanOrEqualTo(threadCount * 95 / 100);
    log.info("并发测试通过: {}/{} 成功", successCount.get(), threadCount);
}
```
