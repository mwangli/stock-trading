```java
@Slf4j
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@Testcontainers
@Transactional
@ActiveProfiles("test")
@TestMethodOrder(MethodOrderer.OrderAnnotation.class)
class FullTradingFlowE2ETest extends BaseIntegrationTest {

    @Container
    static MySQLContainer<?> mysql = new MySQLContainer<>("mysql:8.0")
            .withDatabaseName("stock_trading_test")
            .withInitScript("sql/init-schema.sql");

    @Container
    static MongoDBContainer mongo = new MongoDBContainer("mongo:5.0");

    @DynamicPropertySource
    static void configureDatabase(DynamicPropertyRegistry registry) {
        registry.add("spring.datasource.url", mysql::getJdbcUrl);
        registry.add("spring.datasource.username", mysql::getUsername);
        registry.add("spring.datasource.password", mysql::getPassword);
        registry.add("spring.data.mongodb.uri", mongo::getReplicaSetUrl);
    }

    @Autowired
    private StockInfoMapper stockInfoMapper;

    @Autowired
    private StockPricesRepository stockPricesRepository;

    @Autowired
    private StockSelector stockSelector;

    @Autowired
    private DecisionEngine decisionEngine;

    @Autowired
    private RiskControlService riskControlService;

    @Autowired
    private TradeExecutionService tradeService;

    @Autowired
    private JdbcTemplate jdbcTemplate;

    @Autowired
    private MongoTemplate mongoTemplate;

    // ==================== 阶段1: 数据准备 ====================

    @Test
    @Order(1)
    void stage1_prepareTestData() {
        log.info("=== 阶段1: 准备测试数据 ===");

        // 1.1 插入股票信息到 MySQL
        StockInfo stock = new StockInfo();
        stock.setCode("000001");
        stock.setName("平安银行");
        stock.setMarket("SZ");
        stock.setIsTradable(1);
        stock.setCreateTime(new Date());
        stock.setDeleted("0");
        stockInfoMapper.insert(stock);

        // 验证写入
        assertDataExists("stock_info", "code = '000001'");
        log.info("V1.1 通过: 股票信息写入MySQL");

        // 1.2 插入历史价格到 MongoDB
        for (int i = 0; i < 60; i++) {
            StockPrices price = new StockPrices();
            price.setCode("000001");
            price.setDate(LocalDate.now().minusDays(60 - i));
            price.setPrice1(10.0 + Math.random());  // 开盘
            price.setPrice2(10.5 + Math.random());  // 最高
            price.setPrice3(9.8 + Math.random());   // 最低
            price.setPrice4(10.2 + Math.random());  // 收盘
            price.setVolume(1000000L);
            stockPricesRepository.save(price);
        }

        // 验证写入
        assertMongoRowCount("stockPrices", 60);
        log.info("V1.2 通过: 历史价格写入MongoDB");
    }

    // ==================== 阶段2: 选股流程 ====================

    @Test
    @Order(2)
    void stage2_stockSelection() {
        log.info("=== 阶段2: 选股流程 ===");

        // 2.1 执行选股
        SelectionResult result = stockSelector.selectBestStock(List.of("000001"));
        
        assertThat(result).isNotNull();
        assertThat(result.getSelectedStock()).isEqualTo("000001");
        assertThat(result.getOverallScore()).isGreaterThanOrEqualTo(0);
        log.info("V2.1 通过: 选股执行成功, 得分: {}", result.getOverallScore());

        // 2.2 验证选股结果
        assertThat(result.getLstmScore()).isNotNull();
        assertThat(result.getSentimentScore()).isNotNull();
        assertThat(result.getDexterScore()).isNotNull();
        log.info("V2.2 通过: 三因子得分完整");
    }

    // ==================== 阶段3: 决策流程 ====================

    @Test
    @Order(3)
    void stage3_decisionMaking() {
        log.info("=== 阶段3: 决策流程 ===");

        // 准备选股结果
        SelectionResult selection = new SelectionResult();
        selection.setSelectedStock("000001");
        selection.setOverallScore(75);

        // 3.1 生成决策
        Decision decision = decisionEngine.makeDecision(selection);
        
        assertThat(decision).isNotNull();
        assertThat(decision.getSignal()).isIn(Signal.BUY, Signal.HOLD);
        log.info("V3.1 通过: 决策生成成功, 信号: {}", decision.getSignal());

        // 3.2 验证决策日志写入MySQL
        assertDataExists("decision_log", "stock_code = '000001'");
        log.info("V3.2 通过: 决策日志写入MySQL");
    }

    // ==================== 阶段4: 风控检查 ====================

    @Test
    @Order(4)
    void stage4_riskControl() {
        log.info("=== 阶段4: 风控检查 ===");

        // 4.1 检查风控状态
        RiskStatus status = riskControlService.getStatus();
        
        assertThat(status).isNotNull();
        assertThat(status.isCanBuy()).isTrue();  // 初始状态应该可以买入
        log.info("V4.1 通过: 风控状态正常, canBuy: {}", status.isCanBuy());

        // 4.2 买入前风控检查
        RiskCheckResult checkResult = riskControlService.checkBeforeBuy("000001");
        assertThat(checkResult.isPassed()).isTrue();
        log.info("V4.2 通过: 买入风控检查通过");
    }

    // ==================== 阶段5: 交易执行 ====================

    @Test
    @Order(5)
    void stage5_tradeExecution() {
        log.info("=== 阶段5: 交易执行 ===");

        // 5.1 构造买入请求
        TradeRequest request = new TradeRequest();
        request.setStockCode("000001");
        request.setQuantity(100);
        request.setPrice(10.5);
        request.setTradeType(TradeType.BUY);

        // 5.2 执行买入
        TradeResult result = tradeService.executeBuy(request);
        
        // 注意: 由于是测试环境，可能使用 Mock 券商
        assertThat(result).isNotNull();
        log.info("V5.1 通过: 买入执行完成, 订单ID: {}", result.getOrderId());

        // 5.3 验证订单写入MySQL
        assertDataExists("trade_order", "stock_code = '000001' AND trade_type = 'BUY'");
        log.info("V5.2 通过: 订单写入MySQL");
    }

    // ==================== 完整流程验证 ====================

    @Test
    @Order(6)
    void fullFlow_completeVerification() {
        log.info("=== 完整流程验证 ===");

        // 验证所有数据都已正确写入
        // 1. MySQL 验证
        Integer stockCount = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM stock_info WHERE code = '000001'",
            Integer.class
        );
        assertThat(stockCount).isGreaterThan(0);
        log.info("V6.1 通过: MySQL股票数据存在");

        // 2. MongoDB 验证
        long priceCount = mongoTemplate.getCollection("stockPrices").countDocuments();
        assertThat(priceCount).isGreaterThan(0);
        log.info("V6.2 通过: MongoDB价格数据存在");

        // 3. 决策日志验证
        Integer decisionCount = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM decision_log WHERE stock_code = '000001'",
            Integer.class
        );
        assertThat(decisionCount).isGreaterThan(0);
        log.info("V6.3 通过: 决策日志存在");

        // 4. 订单验证
        Integer orderCount = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM trade_order WHERE stock_code = '000001'",
            Integer.class
        );
        assertThat(orderCount).isGreaterThan(0);
        log.info("V6.4 通过: 交易订单存在");

        log.info("✅ 全流程穿透测试通过!");
    }
}
```

```java
/**
 * 数据流验证工具
 */
public class DataFlowVerifier {

    private final JdbcTemplate jdbcTemplate;
    private final MongoTemplate mongoTemplate;
    private final RedisTemplate<String, String> redisTemplate;

    /**
     * 验证 MySQL 数据流
     */
    public void verifyMySQLFlow(String table, String code, Map<String, Object> expectedValues) {
        // 1. 验证数据存在
        Integer count = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM " + table + " WHERE code = ?",
            Integer.class, code
        );
        assertThat(count).as("数据应该存在").isGreaterThan(0);

        // 2. 验证字段值
        for (Map.Entry<String, Object> entry : expectedValues.entrySet()) {
            Object actual = jdbcTemplate.queryForObject(
                "SELECT " + entry.getKey() + " FROM " + table + " WHERE code = ?",
                Object.class, code
            );
            assertThat(actual).as("字段 " + entry.getKey()).isEqualTo(entry.getValue());
        }
    }

    /**
     * 验证 MongoDB 数据流
     */
    public void verifyMongoDBFlow(String collection, String code, int expectedCount) {
        Query query = Query.query(Criteria.where("code").is(code));
        long count = mongoTemplate.count(query, collection);
        assertThat(count).as("MongoDB数据数量").isEqualTo(expectedCount);
    }

    /**
     * 验证 Redis 数据流
     */
    public void verifyRedisFlow(String keyPattern, String expectedValue) {
        Set<String> keys = redisTemplate.keys(keyPattern);
        assertThat(keys).as("Redis键应该存在").isNotEmpty();
        
        for (String key : keys) {
            String value = redisTemplate.opsForValue().get(key);
            assertThat(value).as("Redis值").isEqualTo(expectedValue);
        }
    }

    /**
     * 打印数据流验证报告
     */
    public void printVerificationReport(String flowName, List<VerificationPoint> points) {
        log.info("========== {} 数据流验证报告 ==========", flowName);
        for (VerificationPoint point : points) {
            String status = point.isPassed() ? "✅ 通过" : "❌ 失败";
            log.info("  {} - {}: {}", point.getId(), point.getName(), status);
        }
        log.info("=========================================");
    }
}
```

```java
@Test
void testConcurrentTradingFlow() throws InterruptedException {
    int threadCount = 10;
    ExecutorService executor = Executors.newFixedThreadPool(threadCount);
    CountDownLatch latch = new CountDownLatch(threadCount);
    AtomicInteger successCount = new AtomicInteger(0);

    for (int i = 0; i < threadCount; i++) {
        final int index = i;
        executor.submit(() -> {
            try {
                // 执行完整交易流程
                String stockCode = "00000" + (index + 1);
                executeFullTradingFlow(stockCode);
                successCount.incrementAndGet();
            } finally {
                latch.countDown();
            }
        });
    }

    latch.await(60, TimeUnit.SECONDS);
    
    // 验证成功率
    assertThat(successCount.get()).isGreaterThanOrEqualTo(threadCount * 95 / 100);
    log.info("并发测试通过: {}/{} 成功", successCount.get(), threadCount);
}
```
